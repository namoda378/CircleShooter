#### 1. 변수와 연산 

   1. 변수의 이름은 영대소문자로 시작하고, 그 뒤에 공백없이 영대소문자 및 '_'기호와 숫자로 이루어져 있다.
      * apple apple1 Apple Apple_1 AppleOne apple___one 등 

```lua 
   -- local, .. , +-/* 

   local namo1 = 5
   namo1 = 3 + 5 
   print( namo1 )
   -- 출력 : 8 

   local cy = 6 + namo1 
   print( cy )
   -- 출력 : 14 

   local my_dogs_name = "DDuby"
   local his_dogs_name = "Mimi"

   print( my_dogs_name + his_dogs_name  )
   -- 오류 !! 

   print( my_dogs_name .. his_dogs_name  )
   -- 출력 : DDubyMimi
```
